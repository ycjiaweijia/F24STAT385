Datetime object has the base type – double

current_datetime <- now()

as_datetime(1721751333, tz = "America/Chicago")

Exercises of converting other formats to the standard ISO8601 format

d1 <- "January 1, 2010"
d2 <- "2015-Mar-07"
d3 <- "06-Jun-2017"
d4 <- c("August 19 (2015)", "July 1 (2015)")
d5 <- "12/30/14" # Dec 30, 2014
t1 <- "1705"
t2 <- "11:15:10.12 PM"

mdy(d1)
#> [1] "2010-01-01"
parse_date(d1, "%B %d, %Y")
#> [1] "2010-01-01"

ymd(d2)
#> [1] "2015-03-07"
parse_date(d2, "%Y-%b-%e")
#> [1] "2015-03-07"

dmy(d3)
#> [1] "2017-06-06"
parse_date(d3, "%e-%b-%Y")
#> [1] "2017-06-06"

mdy(d4)
#> [1] "2015-08-19" "2015-07-01"
parse_date(d4, "%B %d (%Y)")
#> [1] "2015-08-19" "2015-07-01"

mdy(d5)
#> [1] "2014-12-30"
parse_date(d5, "%m/%e/%y")
#> [1] "2014-12-30"


parse_time(t1, "%H%M")
#> 17:05:00
parse_time(t2, "%I:%M:%OS %p")
#> 23:15:10.12

# Split 517 as 5 (which is hour, using 517 %/% 100) 
# and 17 (which is minute, using 517 %% 100)
make_datetime_100 <- function(year, month, day, time) {
  make_datetime(year, month, day, time %/% 100, time %% 100)
}


flights_dt <- flights %>% 
  filter(!is.na(dep_time), !is.na(arr_time)) %>% 
  mutate(
    dep_time = make_datetime_100(year, month, day, dep_time),
    sched_dep_time = make_datetime_100(year, month, day, sched_dep_time)
  )


flights_dt %>% 
  mutate(wday = wday(dep_time, label = TRUE)) %>% 
  ggplot(aes(x = wday )) +
  geom_bar()
